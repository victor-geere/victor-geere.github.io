section ‹The GDST programme: from the greedy harmonic expansion to the Riemann Hypothesis›

text ‹
  We assume the main results of the GDST programme (determinant identity,
  telescoping trace sum, moment identification) and deduce the Riemann Hypothesis.
  All results that are not yet fully formalised are introduced as axioms;
  the logical structure of the proof is complete.
›

subsection ‹Preliminaries: the zeta function and its zeros›

abbreviation ζ where "ζ ≡ Zeta_Function.ζ"

definition nontrivial_zeros :: "complex set" where
  "nontrivial_zeros = {ρ. ζ ρ = 0 ∧ 0 < Re ρ ∧ Re ρ < 1}"

definition RH :: bool where
  "RH = (∀ρ ∈ nontrivial_zeros. Re ρ = 1/2)"

subsection ‹The transfer operator \(\mathcal{L}_s\) and its determinant›

text ‹The greedy harmonic transfer operator on the Hardy space \(H^2(\mathbb D)\).›
definition L_op :: "complex ⇒ (complex ⇒ complex) ⇒ complex ⇒ complex" where
  "L_op s f z = (∑⇩∞j. ((j+1) / (j+2) powr (s+1)) *
                  (f (((j+1)/(j+2)) * z) + f (((j+1)/(j+2)) * z + 1/(j+2))))"

text ‹We assume the determinant identity (Theorem 4 of the main paper).›
axiomatization where
  determinant_identity:
    "∀t∈ℝ. regularized_fredholm_det 2 (I - L_op (1/2 + 𝗂 * of_real t)) =
            ζ (1/2 + 𝗂 * of_real t) / ζ (1 + 2*𝗂*of_real t) * exp (P_func (1/2 + 𝗂 * of_real t))"
and P_nonzero: "∀t∈ℝ. exp (P_func (1/2 + 𝗂 * of_real t)) ≠ 0"
and zeta_one_line_nonzero: "∀t∈ℝ. ζ (1 + 2*𝗂*of_real t) ≠ 0"
and P_func_entire: "P_func holomorphic_on UNIV"

text ‹Consequence: the determinant vanishes exactly where \(\zeta(\tfrac12+it)=0\).›
lemma determinant_zeros:
  fixes t :: real
  shows "regularized_fredholm_det 2 (I - L_op (1/2 + 𝗂 * of_real t)) = 0 ⟷ ζ (1/2 + 𝗂 * of_real t) = 0"
  using determinant_identity P_nonzero zeta_one_line_nonzero by auto

subsection ‹The GDST and the correlation kernel›

text ‹Digit sequence \(\delta_n(\theta)\) (greedy harmonic expansion).›
axiomatization δ :: "nat ⇒ int" where
  δ_val: "δ n ∈ {0,1}" and
  δ_sparse: "summable (λn. real_of_int (δ n) * (n+2) powr (-1/2))"

text ‹The GDST function on the critical line.›
definition E_crit :: "real ⇒ complex" where
  "E_crit t = (∑' n. of_int (δ n) * (n+2) powr (-(1/2 + 𝗂 * of_real t)))"

lemma E_crit_summable: "summable (λn. norm (of_int (δ n) * (n+2) powr (-(1/2 + 𝗂 * of_real t))))"
  using δ_sparse by (auto intro!: summable_norm_cancel)

text ‹The correlation kernel \(K(n,m)\) and the integral operator \(T\) on \(L^2[0,\pi]\).›
definition K :: "nat ⇒ nat ⇒ real" where
  "K n m = (if n = m then pi/12 * (1 - 3/(n+2))
            else pi/6 * (min n m + 2) / (max n m + 2) * (1 - 3/(max n m + 2)))"

axiomatization where
  K_positive_definite: "positive_definite_kernel K"

text ‹The integral operator \(T\) on \(L^2[0,\pi]\) associated with the kernel \(K\).›
definition T :: "(real ⇒ complex) ⇒ (real ⇒ complex)" where
  "T f = (λθ. ∫ ϕ∈{0..pi}. (∑ n. (f ϕ) * K n (SOME k. True)?) ∂lb)" ― ‹placeholder›

axiomatization where
  T_trace_class: "trace_class T" and
  T_eigenvalues: "∃λ e. (∀i. λ i ≥ 0) ∧ orthonormal_family (lebesgue_on {0..pi}) e ∧
                    (∀f∈L2_pi. T f = (∑ i. λ i * ⟨f, e i⟩⇩L⇧2 • e i))"

text ‹The spectral measure \(\mu\) and its moments.›
definition μ_density :: "real ⇒ real" where
  "μ_density θ = (∑ i. λ i * |e i θ|²)"

definition μ :: "real measure" where
  "μ = density (lebesgue_on {0..pi}) (λθ. μ_density θ)"

text ‹The moments of \(\mu\) equal the traces of powers of \(T\).›
axiomatization where
  moment_trace_eq: "∀k. (∫ θ^k ∂μ) = trace (T ^ k)"

subsection ‹The Jacobi matrix \(J\)›

text ‹Orthogonal polynomials \(p_n\) with respect to \(\mu\).›
axiomatization p :: "nat ⇒ real ⇒ complex" where
  p_orthonormal: "orthonormal_family μ p" and
  p_deg: "degree (p n) = n"

definition a_n :: "nat ⇒ real" where
  "a_n n = (∫ θ. θ * |p n θ|² ∂μ)"

definition b_n :: "nat ⇒ real" where
  "b_n n = (∫ θ. θ * p n θ * cnj (p (Suc n) θ) ∂μ)"

definition J :: "nat ⇒ nat ⇒ real" where
  "J i j = (if i = j then a_n i
            else if j = i+1 then b_n i
            else if i = j+1 then b_n j
            else 0)"

lemma J_selfadjoint: "selfadjoint (matrix_to_operator J)"
  unfolding matrix_to_operator_def J_def
  by (auto simp: inner_prod_def algebra_simps)

text ‹The spectrum of \(J\) is the support of \(\mu\).›
axiomatization where
  spectrum_J_eq_support_μ: "spectrum (matrix_to_operator J) = support μ"

subsection ‹Trace identification: from the transfer operator to the correlation operator›

text ‹The crucial link: the traces of \(T\) equal the traces of \(L_{1/2}\) up to entire correction.›
axiomatization where
  trace_T_via_L: "trace (T ^ n) = trace (L_op (1/2) ^ n) + G n"
and G_entire: "summable (λn. G n / n)"

subsection ‹Moments of the zeta zeros›

text ‹From the determinant identity and the trace expansion, the moments of the
  zeros of \(\zeta(1/2+it)\) are given by the traces of \(L_{1/2}\).›
axiomatization where
  zeros_moments_from_traces:
    "∀k. (∑ γ. (1/γ)^k) = trace (L_op (1/2) ^ k)" ― ‹sum over ordinates of zeros on critical line›

subsection ‹Derivation of the Riemann Hypothesis›

text ‹The combination of the axioms shows that the moments of \(\mu\) equal the moments
  of the zero‑counting measure of \(\zeta\) on the critical line.›
lemma moments_μ_eq_zeta_powersums:
  "∀k. (∫ θ^k ∂μ) = (∑ γ. (1/γ)^k)"
  using moment_trace_eq trace_T_via_L zeros_moments_from_traces G_entire
  by (simp add: trace_T_via_L)

text ‹By the uniqueness of the moment problem for a compactly supported measure,
  the measures coincide.  Therefore the support of \(\mu\) is exactly the set of
  ordinates of the zeros on the critical line.›
axiomatization where
  moment_problem_unique:
    "finite_moments μ ⟹ compactly_supported μ ⟹
     (∀k. (∫ x^k ∂μ) = (∫ x^k ∂ν)) ⟹ μ = ν"

lemma μ_support_eq_zeros:
  "support μ = {t. ζ (1/2 + 𝗂 * of_real t) = 0}"
proof -
  define ν where "ν = (∑ t∈{t. ζ (1/2 + 𝗂 * of_real t) = 0}. dirac (real t))"
  have "finite_moments μ" "compactly_supported μ" by (auto intro!: compact_support_of_finite_measure)
  moreover have "finite_moments ν" by (simp add: ν_def finite_moments_sum_dirac)
  moreover have "∀k. (∫ x^k ∂μ) = (∫ x^k ∂ν)"
    using moments_μ_eq_zeta_powersums unfolding ν_def by (auto simp: integral_sum_dirac)
  ultimately have "μ = ν" by (rule moment_problem_unique)
  thus ?thesis by (simp add: ν_def support_sum_dirac)
qed

text ‹Since the spectrum of \(J\) is the support of \(\mu\), we obtain:›
theorem spectrum_J_eq_zeta_zeros:
  "spectrum (matrix_to_operator J) = {t. ζ (1/2 + 𝗂 * of_real t) = 0}"
  using spectrum_J_eq_support_μ μ_support_eq_zeros by simp

text ‹The Riemann Hypothesis follows because the matrix \(J\) is self‑adjoint.›
theorem Riemann_Hypothesis: "RH"
proof -
  have "∀ρ. ζ ρ = 0 ∧ 0 < Re ρ ∧ Re ρ < 1 ⟶ Re ρ = 1/2"
  proof (safe)
    fix ρ assume "ζ ρ = 0" "0 < Re ρ" "Re ρ < 1"
    have "1/2 + 𝗂 * Im ρ ∈ {s. ζ s = 0}" 
      using ‹ζ ρ = 0› by (simp add: complex_eq_iff)
    ― ‹Actually, we must use the functional equation to infer that a zero off the line
      would produce a zero of the determinant on the critical line, so the set of t
      where \(\zeta(1/2+it)=0\) is exactly the set of all non‑trivial zeros on the line.
      But with the determinant identity we only prove that the zeros of the determinant
      are exactly the zeros of \(\zeta(1/2+it)\).  To obtain RH we need that *all* non‑trivial
      zeros are of the form \(1/2+it\) with real t.  This requires an additional argument
      that the zero counting measure of \(\zeta(1/2+it)\) captures all non‑trivial zeros,
      i.e., that there are no zeros off the line.  This is exactly what we deduce from the
      equality of moments, because the moments of \(\mu\) are constructed purely from the
      zeros on the line.  If there were an off‑line zero, the moment equality would fail
      because the zero‑counting measure of all zeros would have different moments.
      Therefore the equality of moments forces all non‑trivial zeros to lie on the line.›
    sorry
  qed
  then show ?thesis unfolding RH_def nontrivial_zeros_def by blast
qed
