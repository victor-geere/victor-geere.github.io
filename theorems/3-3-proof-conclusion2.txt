lemma moment_problem_unique:

section ‹11. Derivation of the Riemann Hypothesis›

text ‹Now we assemble all the pieces to prove RH.›

lemma μ_support_eq_zeros:
  "support μ = {t. ζ (1/2 + 𝗂 * of_real t) = 0}"
proof -
  define ν where "ν = (∑ t∈{t. ζ (1/2 + 𝗂*of_real t) = 0}. dirac (real t))"
  have "compactly_supported μ" "compactly_supported ν"
    (* using that μ lives on [0,π] and ν is supported on a countable set *)
    sorry
  moreover have "∀k. (∫ x^k ∂μ) = (∫ x^k ∂ν)"
    unfolding ν_def
    using moment_eq_trace trace_T_via_L zeros_moment_from_traces G_sum_summable
    by (auto simp: integral_sum_dirac)  (* needs details *)
  ultimately have "μ = ν" by (rule moment_problem_unique)
  thus ?thesis by (simp add: ν_def support_sum_dirac)
qed

theorem Riemann_Hypothesis: "RH"
proof (unfold RH_def nontrivial_zeros_def, safe)
  fix ρ
  assume "ζ ρ = 0" "0 < Re ρ" "Re ρ < 1"
  have "spectrum (matrix_to_operator J_mat) = support μ"
    by (rule spectrum_J_eq_support_μ)
  also have "… = {t. ζ (1/2 + 𝗂 * of_real t) = 0}"
    by (rule μ_support_eq_zeros)
  finally have "spectrum (matrix_to_operator J_mat) = {t. ζ (1/2 + 𝗂*of_real t) = 0}" .

  ― ‹We now need an argument that every non‑trivial zero gives a point in the spectrum.›
  have "∃t. ρ = 1/2 + 𝗂 * of_real t"
  proof -
    from `ζ ρ = 0` have "ζ (1 - ρ) = 0" by (rule zeta_zero_reflect)
    have "Re ρ = 1/2" (* This is what we are trying to prove! So we need a non‑circular argument. *)
    sorry
    thus ?thesis by (rule complex_eq_iff_Re_Im_eq) auto
  qed
  then obtain t where t: "ρ = 1/2 + 𝗂 * of_real t" by blast
  then have "ζ (1/2 + 𝗂 * of_real t) = 0" using `ζ ρ = 0` by simp
  hence "t ∈ {t. ζ (1/2 + 𝗂*of_real t) = 0}" by blast
  then have "t ∈ spectrum (matrix_to_operator J_mat)" using `spectrum ...` by auto
  from J_mat_selfadjoint have "spectrum (matrix_to_operator J_mat) ⊆ ℝ"
    by (simp add: selfadjoint_imp_real_spectrum)
  hence "t ∈ ℝ" using `t ∈ spectrum ...` by auto
  thus "Re ρ = 1/2" using t by (simp add: complex_eq_iff)
qed
