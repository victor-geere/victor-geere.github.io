section ‹1.3 Correlation kernel of the greedy digits›

text ‹
  The greedy digit functions \(\delta_n(\theta)\) are random variables
  with respect to the uniform measure on \([0,\pi]\).  The correlation
  kernel
  \[
    K(n,m) = \int_0^\pi \bigl(\delta_n(\theta)-\tfrac{\theta}{\pi}\bigr)
                          \bigl(\delta_m(\theta)-\tfrac{\theta}{\pi}\bigr)\,d\theta
  \]
  measures the covariance of the digits around their mean value
  \(\theta/\pi\).  This kernel is positive semi‑definite and has a
  closed form that can be computed explicitly from the threshold angles.
›

subsection ‹Mean value of the digits›

text ‹
  For each index \(n\), the function \(\theta\mapsto\delta_n(\theta)\)
  is an indicator of the interval \([\theta_n^*,\pi]\).  Hence its
  mean value is \((\pi-\theta_n^*)/\pi\).
  On the other hand, from the greedy representation we have
  \[
    \frac{1}{\pi}\int_0^\pi \delta_n(\theta)\,d\theta
    = \frac{1}{\pi}\int_{\theta_n^*}^\pi 1\,d\theta
    = \frac{\pi-\theta_n^*}{\pi},
  \]
  which is **not** equal to \(\theta/\pi\) in general.  However, the
  overall expansion \(\sum \delta_n/(n+2) = \theta/\pi\) shows that the
  weighted average of the digits with weights \(1/(n+2)\) equals
  \(\theta/\pi\).  The centred variables \(f_n(\theta)=\delta_n(\theta)-\theta/\pi\)
  therefore satisfy \(\sum f_n/(n+2)=0\), but each individual mean is
  different.
›

definition f_digit :: "nat ⇒ real ⇒ real" where
  "f_digit n θ = real (δ_θ θ n) - θ / pi"

text ‹We use the threshold angle property to compute integrals.›

lemma δ_integrable: "integrable (lebesgue_on {0..pi}) (λθ. real (δ_θ θ n))"
proof -
  have "(λθ. real (δ_θ θ n)) = indicator {threshold_angle n..pi}"
    apply (rule ext)
    using threshold_angle_property[of n]
    by (auto simp: δ_θ_def indicator_def split: if_splits)
  then show ?thesis
    by (simp add: integrable_indicator)
qed

lemma f_digit_integrable: "integrable (lebesgue_on {0..pi}) (f_digit n)"
  unfolding f_digit_def
  using δ_integrable integrable_diff[OF _ integrable_const[of "λθ. θ/pi"]] by auto

subsection ‹Definition of the correlation kernel›

definition K_corr :: "nat ⇒ nat ⇒ real" where
  "K_corr n m = integral {0..pi} (λθ. f_digit n θ * f_digit m θ)"

text ‹The kernel is obviously symmetric and real.›

lemma K_corr_symmetric: "K_corr n m = K_corr m n"
  unfolding K_corr_def by (simp add: mult.commute)

lemma K_corr_real: "K_corr n m ∈ ℝ"
  unfolding K_corr_def by (intro integral_real) auto

subsection ‹Positive semi‑definiteness›

text ‹
  The kernel is a Gram matrix of the family \(\{f_n\}\) in
  \(L^2([0,\pi])\).  Therefore it is automatically positive
  semi‑definite.
›

theorem K_corr_positive_semidefinite:
  fixes a :: "nat ⇒ real"
  assumes "finite (support a)"  ― ‹only finitely many non‑zero coefficients›
  shows "(∑ n m. a n * a m * K_corr n m) ≥ 0"
proof -
  let ?S = "{n. a n ≠ 0}"
  have fin: "finite ?S" using assms by (auto simp: support_def)
  have "(∑ n m. a n * a m * K_corr n m) =
        (∑ n∈?S. ∑ m∈?S. a n * a m * K_corr n m)"
    by (intro suminf_finite) (auto simp: fin)
  also have "… = integral {0..pi} (λθ. (∑ n∈?S. a n * f_digit n θ)^2)"
  proof -
    have "… = (∑ n∈?S. ∑ m∈?S. a n * a m * (∫ θ. f_digit n θ * f_digit m θ ∂(lebesgue_on {0..pi})))"
      unfolding K_corr_def using f_digit_integrable
      by (auto simp: integral_mult_right integral_diff)
    also have "… = integral {0..pi} (λθ. (∑ n∈?S. a n * f_digit n θ)^2)"
      by (auto intro!: integral_sum integral_mult_right[symmetric] 
                       sum.cong fin sum_integral_square)
    finally show ?thesis by simp
  qed
  also have "… ≥ 0"
    by (intro integral_nonneg) (auto simp: power2_eq_square)
  finally show ?thesis .
qed

subsection ‹Closed form (from the Geere 2026 correlation paper)›

text ‹
  The correlation paper gives an explicit formula:
  \[
    K(n,m) =
    \begin{cases}
      \frac{\pi}{12}\bigl(1-\frac{3}{n+2}\bigr), & n=m,\\[4pt]
      \frac{\pi}{6}\frac{\min(n,m)+2}{\max(n,m)+2}
      \bigl(1-\frac{3}{\max(n,m)+2}\bigr), & n\neq m.
    \end{cases}
  \]
  We state this as a theorem (its proof involves lengthy but
  elementary calculations with the threshold angles; we accept it
  here as a reference to the published paper).
›
