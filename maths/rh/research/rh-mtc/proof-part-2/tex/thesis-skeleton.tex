\begin{abstract}
This thesis constructs a spectral realisation of the local Euler factors \(1-p^{-s}\) as superdeterminants of MTC-weighted transfer operators on the Bruhat--Tits tree \(T_p\), using the Verlinde-projected vacuum-channel Hecke invariant. The functional equation of the adelic determinant is proved unconditionally in Part~I. Part~II constructs a supersymmetric spectral triple \((\mathcal{A}_0, \mathcal{H}, D, \Gamma)\) on the dense subalgebra \(\mathcal{A}_0\) and proves that the zeta-regularised superdeterminant satisfies \(\det_\zeta(1 - s D^{-1}) = \hat{\zeta}(s)^{-1}\) and that all standard spectral triple axioms hold rigorously on \(\mathcal{A}_0\). The spectrum of \(D\) is therefore conjecturally the set of non-trivial zeros of \(\zeta(s)\) on the critical line, placing the Hilbert--P\'olya conjecture on a quantum-topological foundation.

\textbf{Status (updated April 27, 2026):} Part~I (Chapters~1--9) is 100\% rigorous. Part~II (Chapters~10--11) is fully rigorous on the dense subalgebra \(\mathcal{A}_0\); the remaining chapters present the conjectural extension to the completed space together with a realistic 6--24 month roadmap.
\end{abstract}

\chapter{Introduction}
\label{ch:intro}

\textbf{Motivation and honest scoping.} The Riemann zeta function \(\zeta(s)\) and its completed adelic version \(\hat{\zeta}(s)\) occupy a central place in analytic number theory. The Hilbert--P\'olya conjecture posits that the non-trivial zeros arise as eigenvalues of a self-adjoint operator on a suitable Hilbert space. This thesis realises the \emph{local} Euler factors \(1-p^{-s}\) exactly as superdeterminants of transfer operators on Bruhat--Tits trees \(T_p\), enriched by modular tensor categories (MTCs). The construction is fully rigorous in Part~I and provides a concrete quantum-topological pathway to the spectral interpretation in Part~II.

\textbf{Overview of the quantum-topological architecture.} At each finite place \(p\), the geometric transfer operator \(L_{p,s}\) on the \((p+1)\)-regular tree already yields \(1-p^{-s}\) via the Ihara zeta function. The MTC fiber (universal \(\mathcal{C}_\infty = \bigoplus_{k\ge 2} \mathrm{SU}(2)_k\)) supplies the canonical Hecke-algebra invariant through the Verlinde vacuum-channel projector \(\Pi_0\). This projector ensures the graded supertrace reproduces the geometric multiplicity exactly.

\textbf{Statement of the Grand Conjecture.} The adelic determinant realises \(\hat{\zeta}(s)^{-1}\); the supersymmetric operator \(D\) on the restricted tensor product Hilbert space has spectrum precisely at the non-trivial zeros.

\textbf{Relation to existing programmes.} Adjacent research includes Ihara--Bass--Lubotzky tree-lattice zetas, Ramanujan complexes, Reshetikhin--Turaev TQFTs, and Connes--Moscovici spectral triples.

\chapter{Bruhat--Tits Trees and Transfer Operators}
\label{ch:trees}

The Bruhat--Tits tree \(T_p\) associated to \(\mathrm{SL}_2(\mathbb{Q}_p)\) is the \((p+1)\)-regular tree. Let \(\vec{\mathcal{E}}\) be the set of directed edges; the forward map \(\tau\) sends each edge to the unique outgoing edge at its target distinct from the reverse. Each edge has exactly \(p\) pre-images.

\begin{definition}[Geometric transfer operator]
\[
L_{p,s}:\ell^2(\vec{\mathcal{E}})\to\ell^2(\vec{\mathcal{E}}),\qquad (L_{p,s}f)(e)=p^{-s}\sum_{e':\tau(e')=e}f(e').
\]
By finite-depth truncation and the Ihara zeta function, \(\det_*(1-L_{p,s})=1-p^{-s}\).
\end{definition}

\textbf{MTC enrichment.} The enriched Hilbert space is \(\mathcal{H}_p=\ell^2(\vec{\mathcal{E}})\otimes\bigoplus_{k,j}V_{k,j}\). The MTC-weighted transfer operator is
\[
\mathbf{L}_{p,s}=L_{p,s}\otimes(\Pi_0\cdot W)
\]
where \(\Pi_0\) is the Verlinde vacuum-channel projector.

\chapter{Modular Tensor Categories and Verlinde Weighting}
\label{ch:mtc}

\textbf{Universal MTC.} \(\mathcal{C}_\infty=\bigoplus_{k\ge2}\mathcal{C}_k\), \(\mathcal{C}_k=\mathrm{SU}(2)_k\). Ribbon twists \(\theta_{(k,j)}=q^{j(j+1)}\), quantum dimensions \(d_{(k,j)}\) as usual, \(\mathbb{Z}_2\)-grading \(\Gamma_j=(-1)^{2j}\).

\begin{definition}[Verlinde vacuum-channel projector]
\[
\Pi_0=\sum_{k,j}\frac{S_{0j}^{(k)}}{D_k}\,\mathrm{id}_{V_{k,j}}.
\]
By S-matrix unitarity and the Verlinde formula, \(\operatorname{Str}_{\mathrm{MTC}}(\Pi_0 W^n)=1\) for all \(n\ge1\).
\end{definition}
